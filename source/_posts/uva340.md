typedef pair <int,int> pii;

// LLONG_MIN LLONG_MAX INT_MIN INT_MAX
const int MAX_N = 1010;


int a[MAX_N], b[MAX_N];

void init(){
    
}

int main(){
    int n, TC = 0;
    while(scanf("%d", &n) != EOF){
        if(n == 0) break;
        TC++;
        printf("Game %d:\n", TC);
        for(int i = 0; i < n; i++) scanf("%d", &a[i]);
        for(;;){
            int A = 0, B = 0;
            for(int i = 0; i < n; i++){
                scanf("%d", &b[i]);
                if(a[i] == b[i]) A++;
            }
            if(b[0] == 0) break;
            for(int k = 1; k <= 9; k++){
                int c1 = 0, c2 = 0;
                for(int i = 0; i < n; i++){ // 紀錄 k 在答案和猜測中出現幾次
                    if(a[i] == k) c1++;
                    if(b[i] == k) c2++;
                }
                if(c1 < c2) B += c1;
                else B += c2;
            }
            printf("    (%d,%d)\n", A, B-A);
        }
    }
    return 0;
}
