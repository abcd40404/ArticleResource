        for(int i = 1; i <= len; i++){
            if(len % i == 0){
                string tmp = s.substr(0, i);
                bool ok = true;
                for(int j = 0; j + i <= len; j += i){
                    if(s.substr(j, i) != tmp){

                    }
                }
                if(ok){

                    break;
                }
            }
        }
